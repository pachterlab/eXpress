00322 void project_to_polytope(vector<Transcript*> bundle_trans, vector<double>& trans_counts, double bundle_counts)
00323 {
00324     vector<bool> polytope_bound(bundle_trans.size(), false);
00325     
00326     while (true)
00327     {
00328         double unbound_counts = 0;
00329         double bound_counts = 0;
00330         for (size_t i = 0; i < bundle_trans.size(); ++i)
00331         {
00332             Transcript& trans = *bundle_trans[i];
00333             
00334             if (trans_counts[i] > trans.tot_counts())
00335             {
00336                 trans_counts[i] = trans.tot_counts();
00337                 polytope_bound[i] = true;
00338             }
00339             else if (trans_counts[i] < trans.uniq_counts())
00340             {
00341                 trans_counts[i] = trans.uniq_counts();
00342                 polytope_bound[i] = true;
00343             }
00344             
00345             if (polytope_bound[i])
00346             {
00347                 bound_counts += trans_counts[i];
00348             }
00349             else
00350             {
00351                 unbound_counts += trans_counts[i];
00352             }
00353         }
00354         
00355         if (unbound_counts + bound_counts == bundle_counts)
00356             return;
00357         
00358         double normalizer = (bundle_counts - bound_counts)/unbound_counts;
00359         bool unbound_exist = false;
00360         unbound_counts = 0;
00361         for (size_t i = 0; i < bundle_trans.size(); ++i)
00362         {    
00363             if (!polytope_bound[i])
00364             {
00365                 trans_counts[i] *= normalizer;
00366                 unbound_counts += trans_counts[i];
00367                 unbound_exist = true;
00368             }
00369         }
00370         
00371         if (unbound_counts + bound_counts - bundle_counts < EPSILON)
00372             return;
00373         
00374         if (!unbound_exist)
00375             polytope_bound = vector<bool>(bundle_trans.size(), false);
