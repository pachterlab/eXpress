00063 void SequenceFwd::set(const std::string& seq, bool rev) {
00064   char* ref_seq = new char[seq.length()];
00065   for (size_t i = 0; i < seq.length(); i++) {
00066     ref_seq[i] = (rev) ? complement(ctoi(seq[seq.length()-1-i])) : ctoi(seq[i]);
